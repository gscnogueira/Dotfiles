** Don't execute Neovim inside Neovim!

Avoids to invoke the neovim command inside a buffer in terminal mode

#+begin_src vimrc
if has('nvim') && executable('nvr')
  let $VISUAL="nvr -cc split --remote-wait +'set bufhidden=wipe'"
endif
#+end_src
